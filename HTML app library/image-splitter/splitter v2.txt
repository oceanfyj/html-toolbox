<script>
    document.addEventListener('DOMContentLoaded', () => {
        // --- Element References ---
        const fileInput = document.getElementById('fileInput');
        const dropZone = document.getElementById('dropZone');
        const sourceImage = document.getElementById('sourceImage');
        const appControls = document.getElementById('appControls');
        const gridOverlay = document.getElementById('gridOverlay');
        const downloadBtn = document.getElementById('downloadBtn');
        const statusText = document.getElementById('statusText');
        const customColsInput = document.getElementById('customCols');
        const customRowsInput = document.getElementById('customRows');

        let currentFile = null;
        let currentCols = 0;
        let currentRows = 0;

        // --- Event Listeners for Buttons (Avoiding inline onclick) ---
        document.getElementById('btn-2x1').onclick = () => applyPreset(2, 1);
        document.getElementById('btn-2x2').onclick = () => applyPreset(2, 2);
        document.getElementById('btn-3x2').onclick = () => applyPreset(3, 2);
        document.getElementById('btn-4x2').onclick = () => applyPreset(4, 2);
        document.getElementById('btn-5x2').onclick = () => applyPreset(5, 2);
        document.getElementById('btn-custom').onclick = () => applyCustom();
        downloadBtn.onclick = () => processAndDownload();

        // --- Drag & Drop Logic ---
        // Prevent default browser behavior for drag events
        ['dragenter', 'dragover', 'dragleave', 'drop'].forEach(eventName => {
            dropZone.addEventListener(eventName, preventDefaults, false);
            document.body.addEventListener(eventName, preventDefaults, false);
        });

        function preventDefaults(e) {
            e.preventDefault();
            e.stopPropagation();
        }

        // Highlight drop zone
        ['dragenter', 'dragover'].forEach(eventName => {
            dropZone.addEventListener(eventName, () => dropZone.classList.add('drag-active'), false);
        });

        ['dragleave', 'drop'].forEach(eventName => {
            dropZone.addEventListener(eventName, () => dropZone.classList.remove('drag-active'), false);
        });

        // Handle Drop
        dropZone.addEventListener('drop', (e) => {
            const dt = e.dataTransfer;
            const files = dt.files;
            handleFiles(files);
        });

        // Handle Manual Selection
        fileInput.addEventListener('change', (e) => {
            handleFiles(e.target.files);
        });

        function handleFiles(files) {
            if (files && files[0]) {
                currentFile = files[0];
                const reader = new FileReader();
                reader.onload = function(event) {
                    sourceImage.src = event.target.result;
                    sourceImage.onload = () => {
                        appControls.style.display = 'block';
                        // Wait slightly for layout to settle
                        setTimeout(() => {
                            window.scrollTo({ top: dropZone.offsetTop + 100, behavior: 'smooth' });
                        }, 100);
                        resetState();
                    };
                }
                reader.readAsDataURL(currentFile);
            }
        }

        function resetState() {
            gridOverlay.innerHTML = '';
            downloadBtn.disabled = true;
            statusText.textContent = "Image Loaded. Choose a split option.";
            currentCols = 0;
            document.querySelectorAll('.btn-group button').forEach(b => b.classList.remove('active'));
        }

        // --- Split Logic ---
        function applyPreset(cols, rows) {
            setSplit(cols, rows);
            // Visual feedback
            document.querySelectorAll('.btn-group button').forEach(btn => {
                btn.classList.remove('active');
                if(btn.id === `btn-${cols}x${rows}`) {
                    btn.classList.add('active');
                }
            });
            customColsInput.value = '';
            customRowsInput.value = '';
        }

        function applyCustom() {
            const c = parseInt(customColsInput.value);
            const r = parseInt(customRowsInput.value);
            if (c > 0 && r > 0) {
                document.querySelectorAll('.btn-group button').forEach(b => b.classList.remove('active'));
                setSplit(c, r);
            } else {
                alert("Please enter valid numbers.");
            }
        }

        function setSplit(cols, rows) {
            currentCols = cols;
            currentRows = rows;
            statusText.textContent = `Ready to split into ${cols} x ${rows} grid`;
            downloadBtn.disabled = false;
            
            gridOverlay.style.gridTemplateColumns = `repeat(${cols}, 1fr)`;
            gridOverlay.style.gridTemplateRows = `repeat(${rows}, 1fr)`;
            gridOverlay.innerHTML = '';

            for (let i = 0; i < cols * rows; i++) {
                const cell = document.createElement('div');
                cell.className = 'grid-cell';
                gridOverlay.appendChild(cell);
            }
        }

        // --- Download Logic ---
        async function processAndDownload() {
            if (!currentFile || currentCols === 0) return;

            const originalText = downloadBtn.textContent;
            downloadBtn.textContent = "Processing...";
            downloadBtn.disabled = true;

            try {
                const zip = new JSZip();
                const img = sourceImage;
                const originalWidth = img.naturalWidth;
                const originalHeight = img.naturalHeight;

                // 1. Math.floor ensures we get integers (no half pixels)
                const pieceWidth = Math.floor(originalWidth / currentCols);
                const pieceHeight = Math.floor(originalHeight / currentRows);

                // 2. Calculate offset to center the cuts
                const totalUsedWidth = pieceWidth * currentCols;
                const totalUsedHeight = pieceHeight * currentRows;
                const startOffsetX = Math.floor((originalWidth - totalUsedWidth) / 2);
                const startOffsetY = Math.floor((originalHeight - totalUsedHeight) / 2);

                const format = currentFile.type; 
                const ext = currentFile.name.split('.').pop();
                const nameBase = currentFile.name.substring(0, currentFile.name.lastIndexOf('.'));

                for (let r = 0; r < currentRows; r++) {
                    for (let c = 0; c < currentCols; c++) {
                        const canvas = document.createElement('canvas');
                        canvas.width = pieceWidth;
                        canvas.height = pieceHeight;
                        const ctx = canvas.getContext('2d');

                        const srcX = startOffsetX + (c * pieceWidth);
                        const srcY = startOffsetY + (r * pieceHeight);

                        ctx.drawImage(
                            img, 
                            srcX, srcY, pieceWidth, pieceHeight, 
                            0, 0, pieceWidth, pieceHeight
                        );

                        const blob = await new Promise(resolve => canvas.toBlob(resolve, format, 1.0));
                        // 1-indexed naming: row-col
                        zip.file(`${nameBase}_${r+1}-${c+1}.${ext}`, blob);
                    }
                }

                const content = await zip.generateAsync({type:"blob"});
                saveAs(content, `${nameBase}_split.zip`);
                downloadBtn.textContent = originalText;
                downloadBtn.disabled = false;

            } catch (err) {
                console.error(err);
                alert("An error occurred during processing. Please try again.");
                downloadBtn.textContent = originalText;
                downloadBtn.disabled = false;
            }
        }
    });
</script>
